class Solution:
    def uniqueMorseRepresentations(self, words):
       
        morse = [".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",
                 ".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",
                 ".--","-..-","-.--","--.."]
        return len(set(''.join(morse[ord(ch) - ord('a')]for ch in word) for word in words))
